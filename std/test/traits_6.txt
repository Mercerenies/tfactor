
trait HasType
  type t
end

trait A
  mod B { T1 : HasType }
    mod C { T2 : HasType }
      fun do-swap ( T1.t T2.t -- T2.t T1.t )
    end
  end
end

mod AA
  require A
  mod B { S1 : HasType }
    mod C { T1 : HasType }
      fun do-swap ( S1.t T1.t -- T1.t S1.t )
        Prelude.swap
      end
    end
  end
end
